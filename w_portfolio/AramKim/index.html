/*** Tab Menu ***/

var tabMenu = $(".notice"); // 기준점 설정
                                            
tabMenu.find("ul > li > ul").hide();

// 기본적으로 active가 붙어있는 공지1만 눈에 보이도록
tabMenu.find("ul > li.active > ul").show(); 
                                            
function tabList(e){
    e.preventDefault();
    var target = $(this);
    target.next().show().parent("li").addClass("active")
        .siblings("li").removeClass("active").find("ul").hide();
    // 버튼(a)을 클릭하면 옆에 있는 형제(ul)를 보여주고
    // 형제의 부모(li)를 찾아 클래스에 active를 추가
    // 부모(li)의 형제(다른 li)를 찾아 클래스에 active를 제거
    // 부모의 형제(다른 li)의 자식(ul)을 찾아 숨김
}

// a(버튼)를 찾아 누르고 tabList 함수 실행
tabMenu.find("ul > li > a").click(tabList).focus(tabList);
