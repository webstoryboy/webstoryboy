기사 기출 문제

{
    let data = [10, 6, 7, 9, 3];
    let temp;

    for(let i=0; i<4; i++){
        for(let j=i+1; j<5; j++){
            if(data[i] > data[j]){
                temp = data[i];
                data[i] = data[j];
                data[j] = temp;
            }
        }
    }

    console.log(data)
    //[3, 6, 7, 9, 10]
}
