01. 변수 : 변수에 저장된 데이터 불러옵니다.

{
    //let x = 100;
    //let y = 200;
    //let z = "javascript";

    let x = 100, y = 200, z = "javascript";  //한줄로 표현 가능

    console.log(x);
    console.log(y);
    console.log(z);
}

04. 배열 : 데이터 불러오기 : 2차 배열

2차 배열은 배열 안에 배열이 들어있는 형태를 말합니다. 다음은 2차 배열에 저장된 값을 불러오는 방법입니다. 

{
    const arr = [100, 200, ["javascript", "jquary"]];   

    console.log(arr[0]);
    console.log(arr[1]);
    console.log(arr[2][0]);
    console.log(arr[2][1]);
}
결과 확인하기
100 200 javascript jquary

06. 배열 : 데이터 불러오기 : for()문

배열을 값을 반복문 for문을 통해 불러 올 수 있습니다. 

{
    const arr = [100, 200, 300, 400, 500];
    
    // console.log(arr[0]);
    // console.log(arr[1]);
    // console.log(arr[2]);
    // console.log(arr[3]);
    // console.log(arr[4],);

    //for(초기값, 조건식, 증감식)
    for(let i=0; i<arr.length; i++){   
        console.log(arr[i]);
    }
}
결과 확인하기
100 200 300 400 500

08. 배열 : 데이터 불러오기 : forEach()

for문을 변형한 forEach() 메서드 사용방법입니다. 

{
    const num = [100, 200, 300, 400, 500];

    //for문을 이용해서 출력
    for(let i = 0; i < num.length; i++){
        console.log(num[i]);
    }

    //forEach()를 이용해서 출력
    num.forEach(function(el){
        console.log(el);
    });

    //forEach() 3가지 인자(파라미터)값
    num.forEach(function(element, index, array){
        console.log(element);    // 배열 안의 값 하나 출력
        console.log(index);      // 배열의 번호 출력
        console.log(array);      // 배열 안의 데이터 전부 출력
    });
}

11. 배열 : 데이터 불러오기 : map()

map() 메서드는 배열의 모든 요소에 대해 주어진 함수를 호출하고, 그 결과를 새로운 배열로 반환합니다. 즉, map() 메서드는 배열의 각 요소를 변형하여 새로운 배열을 만듭니다.

{
    const arr = [100, 200, 300, 400, 500];

    //forEach()
    arr.forEach(function(el){
        console.log(el);    
    });

    //map()
    arr.map(function(el){
        console.log(el);
    });

    arr.map(function(element, index, array){
        console.log(element);
        console.log(index);
        console.log(array);
    });
}

12. 배열 : 데이터 불러오기 : 배열 펼침연산자(Spread Operator)

펼침 연산자(Spread operator)는 자바스크립트에서 배열, 객체, 문자열 등의 요소를 펼쳐서 개별 요소로 분리하는 연산자입니다.

{
    let arr = [100, 200, 300, 400, 500];

    console.log(arr);
    console.log(...arr);
}
결과 확인하기
[100, 200, 300, 400, 500]
100 200 300 400 500 
{
    let arr1 = [100, 200, 300, 400, 500];
    let arr2 = [600, 700];

    console.log(...arr1, ...arr2);
}
결과 확인하기
100 200 300 400 500 600 700

13. 배열 : 데이터 불러오기 : 배열 구조분해할당(Destructuring assignment)

배열 구조 분해 할당(Array destructuring assignment)은 배열의 요소를 개별 변수에 할당하는 방법 중 하나입니다. 이를 통해 배열의 각 요소를 개별 변수로 분리하여 사용할 수 있습니다.

{
    let a, b, c;

    [a, b, c] = [100, 200, "javascript"];

    console.log(a)
    console.log(b)
    console.log(c)
}
결과 확인하기
100 200 javascript
